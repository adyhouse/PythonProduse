    pl.product_id,
    pl.sku,
    CASE WHEN p.ID IS NULL THEN 'ORPHANED' ELSE 'OK' END as Status,
    COUNT(*) as Duplicates

GROUP BY pl.product_id, pl.sku
HAVING COUNT(*) > 1 OR p.ID IS NULL
ORDER BY pl.product_id DESC;
